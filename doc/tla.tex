\subsection{Math}
FMHAKernel之前过程
\begin{eqnarray}
    Q &=& X_q \cdot W_Q \in R^{s_{qo} \times \left(h_q \times d_{qk}\right)}\nonumber\\
    K &=& X_k \cdot W_K \in R^{s_{kv} \times \left(h_{kv} \times d_{qk}\right)}\nonumber\\
    V &=& X_v \cdot W_V \in R^{s_{kv} \times \left(h_{kv} \times d_{vo}\right)}\nonumber
\end{eqnarray}
$X_{q, k, v} \in R^{n_{tok}\times l_{emb}}$, $n_{tok}$: token个数, $l_{emb}$为tokenizer生成的embedding向量长度, 这是第一个transformer block, $s_{qo} == s_{kv}$, 之后此关系不存在, 一般$s_{qo} \le s_{kv}$(KV会随着output tok增长)

FMHA计算内容
\begin{eqnarray}
    S &=& Q \cdot K^T\\
    P &=& softmax(S * scale) \\
    O &=& P \cdot V
\end{eqnarray}

\begin{enumerate}
    \item $Q_{s_{qo}\times d_{qk} \times h_q \times b}$.

	$s_{qo}$: seq len of Qury/Output, 当前step要算的token数.

	$d_{qk}$: 每个head内Q/K向量的维度

	$h_q$: Q head number, 

	$b$: batch

	mem layout: $\left[b\right]\left[h_q\right]\left[s_{qo}\right]\left[d_{qk}\right]$
    \item $K_{s_{kv}\times d_{qk}\times h_{kv}\times b}$

	$s_{kv}$: seq len of Key/Value, 被attend的上下文长度

	$h_{kv}$: KV head num
	
	mem layout: $\left[b\right]\left[h_{kv}\right]\left[s_{kv}\right]\left[d_{qk}\right]$
    \item $V_{d_{vo}\times s_{kv}\times h_{kv}\times b}$

	$d_{vo}$: vector len of within V/O heads

	mem layout: $\left[b\right]\left[h_{kv}\right]\left[s_{kv}\right]\left[d_{qk}\right]$
    \item $O_{s_{qo}\times d_{vo}\times h_q\times b}$
	
	mem layout: $\left[b\right]\left[h_{q}\right]\left[s_{qo}\right]\left[d_{vo}\right]$
\end{enumerate}
