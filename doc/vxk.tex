\subsection{build流程}
参照vllm-xpu-kernels/build.sh. 
\begin{lstlisting}
bash build.sh &> /tmp/cmake.log #详细分析, cflags, lflags
make VERBOSE=1 &> /tmp/make.log #build过程详细分析
\end{lstlisting}

需要注意的点:
\begin{enumerate}
    \item AOT(Ahead Of Time) or JIT(Just In Time)?
\begin{lstlisting}

\begin{lstlisting}
-device pvc,bmg,bmg-g21-a0,bmg-g31-a0,须添加ptl
\end{lstlisting}
CMakeLists.txt
\begin{lstlisting}
line:53 option(VLLM_XPU_ENABLE_XE2 "Enable XE2 architecture kernels" ON) #须添加XE3支持
line:182  set(XE2_AOT_DEVICES "pvc,bmg,bmg-g21-a0,bmg-g31-a0")# 须添加Xe3支持set(XE2_AOT_DEVICES "pvc,bmg,bmg-g21-a0,bmg-g31-a0,ptl")
line:189  if(DEFINED ENV{VLLM_XPU_XE2_AOT_DEVICES}) #环境变量override
line:287  DPCPP_SYCL_TARGET #target加上ptl
line:346  #需要增加Xe3
\end{lstlisting}
